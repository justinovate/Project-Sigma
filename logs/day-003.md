% Calculating parameters for the fit
mu = mean(data.Length);

% Overlaying the theoretical curve on real data
x = linspace(min(data.Length), max(data.Length), 100);
y = normpdf(x, mu, sigma);
plot(x, y, 'r-', 'LineWidth', 2);
